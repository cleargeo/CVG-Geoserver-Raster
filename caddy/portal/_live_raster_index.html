Connection Examples
Ready-to-use code for Python, JavaScript, QGIS, cURL, and GDAL.
# pip install owslib
from owslib.wms import WebMapService
wms = WebMapService("https://raster.cleargeo.tech/geoserver/wms", version="1.3.0")
print(list(wms.contents.keys()))
img = wms.getmap(layers=["workspace:layer"], srs="EPSG:4326", bbox=(-180,-90,180,90), size=(1024,512), format="image/png")
with open("output.png","wb") as f: f.write(img.read())
// npm install ol
import TileLayer from "ol/layer/Tile"; import TileWMS from "ol/source/TileWMS";
const l = new TileLayer({source:new TileWMS({url:"https://raster.cleargeo.tech/geoserver/wms",params:{LAYERS:"workspace:layer",TILED:true},serverType:"geoserver"})});
